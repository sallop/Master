 \sum_{i=0}^{D} \sum_{j=i+1}^{D} x_{i} W_{ij} x_{j}
  &=&
  \sum_{j=1}^{D} W_{0j} x_{j}         +
  \sum_{i=0  }^{D}
  \sum_{j=i+1}^{D} x_{i} W_{ij} x_{j}\\
 &=&
  \sum_{j=1}^{D} a_{j} x_{j}                       +
  \sum_{i=1}^{D} \sum_{j=i+1}^{D} x_{i} W_{ij} x_{j}\\
\end{eqnarray}

\section{神経回路の更新則}
元となる神経回路網の更新則は
\begin{eqnarray}
 W_{ij}(t+1) &=& W_{ij}(t) + \Delta W_{ij}(t)\\
\end{eqnarray}

ここで$W_{0j}$はバイアス項であり,その更新則は
\begin{eqnarray}
 \Delta W_{0j} &=& \epsilon (x_0       x_j - \hat x_0   \hat x_j)\\
               &=& \epsilon (  1 \cdot x_j -  1   \cdot \hat x_j)\\
               &=& \epsilon (          x_j -            \hat x_j)
		\label{eq:DeltaW}
\end{eqnarray}
となっている.

式(\ref{eq:DeltaW})において,
$x_j$は環境から入力されるパターン$\bx$の素子$j$番目の発火確率$p(x_j =1)$,
$\hat x_j$は回路$\bx$を自由に走らせたとき$j$番目の
発火確率$p(\hat x_j =1)$を表わしている.

したがって、バイアス項$\ba$についての更新則は式(\ref{eq:DeltaW})より、
\begin{eqnarray}
 \Delta a_{j}  &=& \epsilon (x_j - \hat x_j)\\
\end{eqnarray}
で表わされる.
