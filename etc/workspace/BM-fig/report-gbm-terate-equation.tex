\section{はじめに}
参考文献に書かれている数式の添字に誤りと思われる箇所がある。
\begin{eqnarray}
 p(h_j=1 | \bv, \bh_{-j}) &=&
  \sigma(
    \sum^{D}_{i=1            } W_{ij} v_i    
  + \sum^{P}_{m=1\backslash j} J_{jm} h_j\label{eq:pushup}\\
 p(v_i=1 | \bv, \bv_{-i}) &=&
  \sigma(
    \sum^{P}_{j=1} W_{ij} h_j
  + \sum_{D}_{k=1\backslash i} L_{ik} v_j
  )\label{eq:pushdown}\\
  \mu_j &\leftarrow&
   \sigma( \sum_i W_{ij} v_i + \sum_{m\backslash j} J_{mj} \mu_m )
   \label{eq:meanfield}
\end{eqnarray}

式(\ref{eq:pushup})は
\begin{table}
 \begin{tabular}{|c|c|}[htbp]
  \hline
  誤    & $\sum_{m\backslash j}^{P} = J_{jm} h_j$\\
  \hline
  正  & $\sum_{m\backslash j}^{P} = J_{jm} h_m$\\
  \hline
 \end{tabular}
 \begin{tabular}{|c|c|}
  \hline
  誤    & \sum_{k=1\backslash i}^{D} L_{ik} v_j\\
  \hline
  正  &  \sum_{k=1\backslash i}^{D} L_{ik} v_k\\
  \hline
 \end{tabular}
\end{table}

が正しいだろう。

また、論文中ではバイアス項に関して記述されておらず、
このバイアス項がどれ程の影響を持たらすかもわからない。

式(\ref{eq:pushup}),(\ref{eq:pushdown}),(\ref{eq:meanfield})
の添字とバイアス項の有無がどれだけ影響があるか実験して明かにする。

\section{共通事項}
\begin{eqnarray}
 \sigma(x) &=& \frac{1}{1 + \exp(-x)}\\
\end{eqnarray}

\subsection{GBMBiasIterate}
\begin{eqnarray}
 E(\bv,\bh) &=&
  -\sum_{i=1}^{D} a_i v_i -\sum_{j=1}^{P} b_j h_j\\
 &+&\sum_{i=1}^{D} \sum_{k=i+1}^{D} v_i L_{ik} v_k
  +\sum_{j=1}^{P} \sum_{j=i+1}^{P} h_j J_{jm} h_m
  +\sum_{i=1}^{D} \sum_{j=1}^{P}   v_i W_{ij} h_j\\
 \mu_j &\leftarrow&
  \sigma(
  -  b_j
  + \sum_{i=1}^{D} v_i W_{ij}
  + \sum_{m=1\backslash j}^{P} \mu_m J_{mj}
  )\\
 p(h_j=1 | \bv, \bh_{-j}) &=&
  \sigma(
  - b_j + \sum_{m=1\backslash j}^{P} J_{jm} h_m
  )\\
 p(v_i=1 | \bh, \bv_{-i}) &=&
  \sigma(
  - a_i
  + \sum_{j=1}^{P} W_{ij} h_j 
  + \sum_{k=1}^{D} L_{ik} v_k
  )\\
\end{eqnarray}
