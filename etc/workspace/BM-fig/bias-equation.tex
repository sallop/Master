\section{FBM}
\subsection{バイアス無し}
\begin{eqnarray}
 E(\bv) &=& -\sum_{i=1}^{D}\sum_{k=i+1}^{D} v_i L_{ik} v_k \\
 u_i    &=&                \sum_{k=1  }^{D}     L_{ik} v_k \\
\end{eqnarray}

\subsection{バイアス有り(1)}
\begin{eqnarray}
 E(\bv) &=&
   \sum_{i=1}^{D}                      a_i    v_i
  -\sum_{i=1}^{D} \sum_{k=i+1}^{D} v_i L_{ik} v_k\\
 u_i     &=&
  -\sum_{k=1}^{D} a_i    v_k
  +\sum_{k=1}^{D} L_{ik} v_k\\
\end{eqnarray}

\subsection{バイアス有り(2)}
\begin{eqnarray}
 E(\bv) &=& 
  \sum_{i=1}^{D} a_i v_i
  -\sum_{i=1}^{D}\sum_{k=i+1}^{D} v_i L_{ik} v_k\\
 u_i &=&
  -\sum_{k=1}^{D} a_k    v_k
  +\sum_{k=1}^{D} L_{ik} v_k\\
\end{eqnarray}

\subsection{バイアス有り(3)}
\begin{eqnarray}
 E(\bv) &=& 
  \sum_{i=1}^{D} a_i v_i
  -\sum_{i=1}^{D} \sum_{k=i+1}^{D} v_i L_{ik} v_k\\
 u_i &=&
  - \sum_{k=1}^{D} a_i v_i + L_{ik} v_k\\
\end{eqnarray}

\subsection{バイアス有り(4)}
\begin{eqnarray}
 E(\bv) &=& \sum_{i=1}^{D} a_i v_i
  -\sum_{i=1}^{D}\sum_{k=i+1}^{D} v_i L_{ik} v_k\\
 u_i &=& a_i + \sum_{k=1}^{D} L_{ik} v_k\\
\end{eqnarray}

\section{GBM}
\subsection{バイアス有り}
\begin{eqnarray}
 E(\bv,\bh)
  &=& \sum_{i}^D a_i v_i + \sum_{j}^P b_j h_j\\
 &-&\frac{1}{2}\sum_{i,k}^{D} v_i L_{ik} v_k
  - \frac{1}{2}\sum_{j,m}^{P} h_j J_{jm} h_m
  - \sum_{i=1}^{D}\sum_{j=1}^{P} v_i W_{ij} h_j\\
 p(h_j=1|\bv,\bh_{-j})
  &=&
  \sigma(
  - b_j
  + \sum_{i=1            }^{D} v_i W_{ij}
  + \sum_{m=1\backslash m}^{P} J_{jm} h_m
  )\\
 p(v_i=1|\bh,\bv_{-i})
  &=&
  \sigma(
  - a_i
  + \sum_{j=1            }^{P} W_{ij} h_j
  + \sum_{k=1\backslash i}^{D} L_{ik} v_k
  )\\
 \mu_j
  &\leftarrow&
  \sigma(
  - b_j
  + \sum_{i=1            }^{D}   v_i W_{ij}
  + \sum_{m=1\backslash j}^{P} \mu_m J_{mj}
 )\\
\end{eqnarray}

\subsection{バイアス無し}
\begin{eqnarray}
 E(\bv,\bh)
  &=&
  -\frac{1}{2}\sum_{i,k}^{D} v_i L_{ik} v_k
  -\frac{1}{2}\sum_{j,m}^{P} h_j J_{jm} h_m
  -\sum_{i=1}^{D}\sum_{j=1}^{P} v_i W_{ij} h_j\\
 p(h_j=1 | \bv, \bh_{-j} )
  &=&
  \sigma(
  \sum_{i=1            }^{D} v_i W_{ij} +
  \sum_{m=1\backslash j}^{P} J_{jm} h_m
  )\\
 p(v_i=1 | \bh, \bv_{-i} )
  &=&
  \sigma(
  \sum_{j=1            }^{P} W_{ij} h_j +
  \sum_{m=1\backslash i}^{D} L_{im} v_m
  )\\
 \mu_j
  &\leftarrow&
  \sigma(
  \sum_{i=1            }^{D} v_i  W_{ij}  +
  \sum_{m=1\backslash j}^{P}\mu_m J_{mj}
  )\\
\end{eqnarray}
